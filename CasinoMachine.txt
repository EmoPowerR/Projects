    class MachineType
    {
        private int moneyIn;
        
        public int Money
        {
            get { return this.moneyIn; }
            set
            {
                this.moneyIn = value;

        }

        public void PrintMoney()

            Console.WriteLine(this.moneyIn);

        enum fruits
        {
            bananna =2,
            apple,
            cherry
        }

        public int fruitMachine(int bet)
        {
            Random rnd = new Random();
            int[,] array2d = new int[,]
            {
                {rnd.Next(2,5),rnd.Next(2,5),rnd.Next(2,5) },
                {rnd.Next(2,5), rnd.Next(2,5), rnd.Next(2,5) },
                {rnd.Next(2,5), rnd.Next(2,5), rnd.Next(2,5) }
            };
            bool combination = false;
            for (int i = 0; i < 3; i++)
            {
                bool done = true;
                for (int j = 0; j < 3; j++)
                {
                    if (array2d[i, 0] == array2d[i, 1] && array2d[i, 0] == array2d[i, 2] && done)
                    {
                        bet = bet * array2d[i, 0];
                        combination = true;
                        done = false;
                    }
                    fruits fruit = (fruits)array2d[i, j];
                    Console.Write(fruit+" ");
                }
                Console.WriteLine();
            }
            if (combination)
            {
                return bet;
            }
            else
            {
                return 0;
            }

        }
